#include "stdafx.h"
#include <iostream>
using namespace std;

//仮想関数
class Base{
public:
	string name;

};
class D1 :virtual public Base{};
class D2 :virtual public Base{};

class D3 : public D1,public D2{
public:
	void show(){
		cout << name.c_str() << endl;
	}
};


int _tmain(int argc, _TCHAR* argv[])
{
	D3 d3;
	d3.name = "yamada";
	d3.show();
	return 0;
}
