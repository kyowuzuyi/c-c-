#include "stdafx.h"
#include <iostream>
using namespace std;

void g() {	// ここで、例外処理をスローする
	throw runtime_error("error"); // 自分がスローする
}

void f(){

	try{
		for (int i = 0; i < 10; i++){
			for (int j = 0; j < 10; j++){
				cout << "i:" << i << "j:" << j << endl;
				//i=5かつj=5のときにruntime_errorの例外処理を発生させて、i,jのループを中断してください
				if (i == 5 && j == 5){
					g();
				}

			}
		}
	}
	catch (runtime_error& e){
		cerr << e.what() << endl;
	}

}

int _tmain(int argc, _TCHAR* argv[])
{
		f();
	return 0;
}
