#include "stdafx.h"
#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>
using namespace std;
int _tmain(int argc, _TCHAR* argv[])
{
	vector<float> v = { 1, 6, 3, 2 };

	sort(v.begin(), v.end());
	for (auto& x : v) cout << x << endl;

	reverse(v.begin(), v.end());
	for (auto& x : v) cout << x << endl;

	cout << *min_element(v.begin(), v.end()) << endl;
	cout << *max_element(v.begin(), v.end()) << endl;

	return 0;
}
